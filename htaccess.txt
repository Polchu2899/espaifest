# ════════════════════════════════════════════════════════════
# EspaiFest — Redirecciones 301 para GitHub Pages
# NOTA: GitHub Pages NO soporta .htaccess (Apache).
# Este archivo es para referencia si en el futuro se migra
# a un hosting Apache. Para GitHub Pages se usa _redirects
# (Netlify) o la lógica JS de redirección incluida en index.html
# ════════════════════════════════════════════════════════════

Options -Indexes
RewriteEngine On

# ── Forzar HTTPS ──────────────────────────────────────────
RewriteCond %{HTTPS} off
RewriteRule ^(.*)$ https://%{HTTP_HOST}%{REQUEST_URI} [L,R=301]

# ── Eliminar www ──────────────────────────────────────────
RewriteCond %{HTTP_HOST} ^www\.espaifest\.com [NC]
RewriteRule ^(.*)$ https://espaifest.com/$1 [L,R=301]

# ── Páginas de eventos / servicios → secciones SPA ───────
RewriteRule ^fiestas-infantiles-espaifest-martorell/?$ https://espaifest.com/#fiestas [L,R=301]
RewriteRule ^fiestas-infantiles/?$                      https://espaifest.com/#fiestas [L,R=301]
RewriteRule ^fiesta-ninos/?$                            https://espaifest.com/#fiestas [L,R=301]
RewriteRule ^fiestas-ninos/?$                           https://espaifest.com/#fiestas [L,R=301]
RewriteRule ^encuentros-familiares-amigos/?$            https://espaifest.com/#adultos [L,R=301]
RewriteRule ^eventos-especiales/?$                      https://espaifest.com/#adultos [L,R=301]
RewriteRule ^animacion-infantil/?$                      https://espaifest.com/#fiestas [L,R=301]
RewriteRule ^animacion-adultos-espaifest-martorell/?$   https://espaifest.com/#adultos [L,R=301]

# ── Business ──────────────────────────────────────────────
RewriteRule ^business-espaifest-martorell/?$            https://espaifest.com/#business [L,R=301]
RewriteRule ^fotos-business-espaifest-martorell/?$      https://espaifest.com/#business [L,R=301]
RewriteRule ^local-reuniones-empresas-baix-llobregat/?$ https://espaifest.com/#business [L,R=301]

# ── Servicios / Extras ────────────────────────────────────
RewriteRule ^catering/?$                                https://espaifest.com/#reservas [L,R=301]
RewriteRule ^equipamientos/?$                           https://espaifest.com/#business [L,R=301]
RewriteRule ^tarjeta-invitacion-martorell/?$            https://espaifest.com/#fidelizacion [L,R=301]
RewriteRule ^tarjeta_fidelizacion/?$                    https://espaifest.com/#fidelizacion [L,R=301]

# ── Precios ───────────────────────────────────────────────
RewriteRule ^precios_martorell/?$                       https://espaifest.com/#precios [L,R=301]
RewriteRule ^precios_martorell_1/?$                     https://espaifest.com/#precios [L,R=301]
RewriteRule ^precios/?$                                 https://espaifest.com/#precios [L,R=301]

# ── Galería ───────────────────────────────────────────────
RewriteRule ^la-sala/?$                                 https://espaifest.com/#galeria [L,R=301]
RewriteRule ^galeria-local-fiestas-martorell/?$         https://espaifest.com/#galeria [L,R=301]

# ── Reservas ──────────────────────────────────────────────
RewriteRule ^reservar-local-fiesta-martorell/?$         https://espaifest.com/#reservas [L,R=301]
RewriteRule ^solicitud-para-modificar-reserva/?$        https://espaifest.com/#reservas [L,R=301]
RewriteRule ^contacto-alquiler-local-martorell/?$       https://espaifest.com/#reservas [L,R=301]

# ── Blog / Info ───────────────────────────────────────────
RewriteRule ^blog/?$                                    https://espaifest.com/#blog [L,R=301]
RewriteRule ^blog/(.*)$                                 https://espaifest.com/#blog [L,R=301]
RewriteRule ^promociones/?$                             https://espaifest.com/#precios [L,R=301]
RewriteRule ^last-minute/?$                             https://espaifest.com/#precios [L,R=301]
RewriteRule ^preguntas-frecuentes/?$                    https://espaifest.com/#reservas [L,R=301]
RewriteRule ^noticias-martorell/?$                      https://espaifest.com/#blog [L,R=301]

# ── Páginas SEO locales ───────────────────────────────────
RewriteRule ^local-para-fiestas-baix-llobregat/?$       https://espaifest.com/#fiestas [L,R=301]
RewriteRule ^alquilar-local-graduaciones-baix-llobregat/?$ https://espaifest.com/#adultos [L,R=301]
RewriteRule ^servicios-local-fiesta-martorell/?$        https://espaifest.com/ [L,R=301]
RewriteRule ^local-fiestas-martorell/?$                 https://espaifest.com/ [L,R=301]
RewriteRule ^local-fiestas-baix-llobregat/?$            https://espaifest.com/#fiestas [L,R=301]

# ── Legal ─────────────────────────────────────────────────
RewriteRule ^aviso-legal/?$                             https://espaifest.com/aviso-legal.html [L,R=301]
RewriteRule ^condiciones-de-uso/?$                      https://espaifest.com/condiciones.html [L,R=301]
RewriteRule ^politica-de-privacidad-y-cookies/?$        https://espaifest.com/privacidad.html [L,R=301]
RewriteRule ^politica-de-cookies/?$                     https://espaifest.com/cookies.html [L,R=301]

# ── Medidas COVID (página obsoleta) ──────────────────────
RewriteRule ^medidas-contra-el-covid-19-espaifest-martorell/?$ https://espaifest.com/ [L,R=301]

# ── Cualquier otra URL no mapeada → home ─────────────────
RewriteCond %{REQUEST_FILENAME} !-f
RewriteCond %{REQUEST_FILENAME} !-d
RewriteRule ^(.*)$ https://espaifest.com/ [L,R=301]
