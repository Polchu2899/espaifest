# ════════════════════════════════════════════════════════════
# EspaiFest — .htaccess para cPanel / LiteSpeed
# Hosting: Indedmedia (Apache/LiteSpeed compatible)
# ════════════════════════════════════════════════════════════

Options -Indexes
RewriteEngine On

# ── Forzar HTTPS ──────────────────────────────────────────────
RewriteCond %{HTTPS} off
RewriteRule ^(.*)$ https://%{HTTP_HOST}%{REQUEST_URI} [L,R=301]

# ── Canonicalizar a espaifest.com (sin www, sin .cat/.es) ────
RewriteCond %{HTTP_HOST} ^www\.espaifest\.com$ [NC]
RewriteRule ^(.*)$ https://espaifest.com/$1 [L,R=301]

RewriteCond %{HTTP_HOST} ^(www\.)?espaifest\.cat$ [NC]
RewriteRule ^(.*)$ https://espaifest.com/$1 [L,R=301]

RewriteCond %{HTTP_HOST} ^(www\.)?espaifest\.es$ [NC]
RewriteRule ^(.*)$ https://espaifest.com/$1 [L,R=301]

# ── Security Headers ─────────────────────────────────────────
<IfModule mod_headers.c>
  Header always set X-Frame-Options "SAMEORIGIN"
  Header always set X-Content-Type-Options "nosniff"
  Header always set X-XSS-Protection "1; mode=block"
  Header always set Referrer-Policy "strict-origin-when-cross-origin"
  Header always set Permissions-Policy "camera=(), microphone=(), geolocation=()"

  # HSTS — descomentar solo cuando SSL esté confirmado y estable
  # Header always set Strict-Transport-Security "max-age=31536000; includeSubDomains; preload"

  # CSP ajustado para Firebase SDK + Google Fonts + canvas inline
  Header always set Content-Security-Policy "default-src 'self'; script-src 'self' 'unsafe-inline' https://www.gstatic.com https://apis.google.com; style-src 'self' 'unsafe-inline' https://fonts.googleapis.com; font-src https://fonts.gstatic.com; connect-src 'self' https://*.googleapis.com https://*.firebaseio.com https://firestore.googleapis.com https://identitytoolkit.googleapis.com https://securetoken.googleapis.com; img-src 'self' data: blob: https:; frame-src 'none'; object-src 'none';"
</IfModule>

# ── Proteger archivos sensibles ───────────────────────────────
<Files ".htaccess">
  Order allow,deny
  Deny from all
</Files>
<Files ".htpasswd">
  Order allow,deny
  Deny from all
</Files>
